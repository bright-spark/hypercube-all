CREATE TABLE users(user_id int unsigned not null auto_increment primary key,handle varchar(32),email varchar(64),password varchar(32),facebook_id int unsigned,friendster_id int unsigned,opensocial_id int unsigned,image_id int unsigned);CREATE TABLE networks(network_id int unsigned not null auto_increment primary key,from_user_id int unsigned,to_user_id int unsigned,status smallint);CREATE TABLE facebook(facebook_id int unsigned not null auto_increment primary key,fb_sig_user bigint,cube_id int unsigned,widget_id int unsigned,fb_sig_time varchar(32));CREATE TABLE friendster(friendster_id int unsigned not null auto_increment primary key,id bigint,cube_id int unsigned,widget_id int unsigned);CREATE TABLE opensocial(opensocial_id int unsigned not null auto_increment primary key,id varchar(64),cube_id int unsigned,widget_id int unsigned,host varchar(32));CREATE TABLE cubes(cube_id int unsigned not null auto_increment primary key,name varchar(64),user_id int unsigned,image_id int unsigned,serve_count int unsigned);CREATE TABLE widgets(widget_id int unsigned not null auto_increment primary key,code text,title varchar(64),properties text,settings text,user_id int unsigned,cube_id int unsigned,image_id int unsigned,timestamp timestamp default current_timestamp,hidden boolean,hash char(32));CREATE TABLE images(image_id int unsigned not null auto_increment primary key,image blob,width smallint,height smallint);
